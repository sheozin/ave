**Design note — panel nesting:** `#profile-panel` is a *child* of `#user-chip` (not a sibling as the spec's structural diagram implies). This is intentional: `.profile-panel` uses `position:absolute` and the containing `.user-chip` is `position:relative`, so the panel is anchored to the chip's bottom-right corner.
```js
// TIMEZONE HELPERS
function getUtcOffset(tz) {
  try {
    const s = new Date().toLocaleString('en', { timeZone: tz, timeZoneName: 'shortOffset' });
    const m = s.match(/GMT([+-]\d+(?::\d+)?)/);
    return m ? 'UTC' + m[1] : 'UTC+0';
  } catch { return 'UTC'; }
}

function buildTzSelect(selectId, currentTz) {
  const sel = document.getElementById(selectId);
  if (!sel) return;
  const zones = Intl.supportedValuesOf('timeZone');
  const groups = {};
  zones.forEach(function(tz) {
    var region = tz.includes('/') ? tz.split('/')[0] : 'Other';
    if (!groups[region]) groups[region] = [];
    groups[region].push(tz);
  });
  var defaultTz = currentTz || Intl.DateTimeFormat().resolvedOptions().timeZone;
  var html = '';
  Object.keys(groups).sort().forEach(function(region) {
    html += '<optgroup label="' + region + '">';
    groups[region].forEach(function(tz) {
      var offset = getUtcOffset(tz);
      var selected = tz === defaultTz ? ' selected' : '';
      html += '<option value="' + tz + '"' + selected + '>' + tz + ' (' + offset + ')</option>';
    });
    html += '</optgroup>';
  });
  sel.innerHTML = html;
}
```

Note on safety: all content comes from `Intl.supportedValuesOf('timeZone')` (browser-internal list) and `getUtcOffset()` which uses `toLocaleString`. No user-controlled strings are interpolated.
